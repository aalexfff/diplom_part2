Задание 1

SELECT c.login AS "Логин", COUNT(o."inDelivery") AS "Количество заказов" FROM "Orders" AS o INNER JOIN "Couriers" AS c on o."courierId" = c.id WHERE o."inDelivery" = 't' GROUP BY c.login;
