Задание 1

SELECT c.login AS "Логин", COUNT(o."inDelivery") AS "Количество заказов" FROM "Couriers" AS c INNER JOIN "Orders" AS o on c.id = o."courierId" WHERE o."inDelivery" = 't' GROUP BY c.login;

Задание 2

SELECT track AS "Трэк заказа", CASE WHEN finished = 'true' THEN '2' WHEN cancelled = 'true' THEN '-1' WHEN "inDelivery" = 'true' THEN '1' ELSE '0' END FROM "Orders";
